Using Terraform shared credentials file in aws provider

providers.tf: 

terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
      version = "~> 3.0"
    }
  }
}

provider "aws" {
  region                  = "us-east-1"
  shared_credentials_file = "~/.aws/credentials" # or another file
  profile                 = "default"
}
