Reading an Excel file using Python and xlrd library

import xlrd

# Give the location of the file
loc = ("Input_sheet.xls")

# To open Workbook
wb = xlrd.open_workbook(loc)
sheet = wb.sheet_by_index(5)

# For row 1 and columns 1-20
for i in range(1,20):
  print(sheet.cell_value(1, i))
