for in bash examples

Example 1 

#!/bin/bash

for i in 1 2 3 4 5
do
	echo "Hello $i times"
done

#!/bin/bash

Example 3 
#!/bin/bash

for i in {1..5}
do
	touch $i
done
Example 4 
#!/bin/bash

i=1
for day in Mon Tue Wed Thu Fri Sat Sun
do
	echo "Weekday $((i++)) is $day"
done
Example 5 
#!/bin/bash

i=1
for username in `awk -F: '{print $1}' /etc/passwd`
do
	echo "Username $((i++)) is $username"
done
