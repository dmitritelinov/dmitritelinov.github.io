tasks/main.yml: 
...
- name: configure
  template:
    src: haproxy.cfg.j2
    dest: /etc/haproxy/haproxy.cfg
    owner: root
    group: root
    mode: 0644
  notify: reload haproxy
...

...

    server {{ server }} {{ value.ip }}:8080 check
{% endfor %}
...
