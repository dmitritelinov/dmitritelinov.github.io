Basic printf/scanf in C

#include <stdio.h>
  
int main()
{
    printf("Your name:");
    char name[32];
    scanf("%s", name);

    printf("Your age:");
    int age;
    scanf("%i", &age);

    printf("Your height(m):");
    float height;
    scanf("%f", &height);

    printf("Your name is %s. Your age is %i. Your height is %fm.\n", name, age, height);

    return 0;
}
