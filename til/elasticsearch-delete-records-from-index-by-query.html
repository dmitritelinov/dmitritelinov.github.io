curl -XPOST 'localhost:9200/${INDEX}/_delete_by_query?pretty' -H 'Content-Type: application/json' -d'
{
  "query" : {
    "bool" : {
      "should": [
          { "match": { "account_number": 265 }},

      ]
    }
  }
