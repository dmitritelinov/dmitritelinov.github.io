case in bash example

#!/bin/bash

echo
echo Please choose an option
echo
echo a - Display date and time
echo b - Show kernel version
echo c - List users logged in
echo d - Check system uptime
echo

read option

case $option in
	a) date;;
	b) uname -a;;
	c) who;;
	d) uptime;;
	*) echo Invalid choice - Bye.
esac
